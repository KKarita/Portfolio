Harjoitus 3 - Urheilija-luokka

Tässä tehtävässä luotiin JavaScriptin luokkajärjestelmä ja periytyminen käyttämällä "Henkilö" ja "Urheilija" -luokkia. 

        
        class Person {
            constructor(firstnames, lastname, nickname, yearOfBirth) {
                this.firstnames = firstnames;
                this.lastname = lastname;
                this.nickname = nickname;
                this.yearOfBirth = yearOfBirth;
            }

            getName() {
                return `${this.firstnames} "${this.nickname}" ${this.lastname}`;
            }
        }

        class Athlete extends Person {
            constructor(
              firstnames,
              lastname,
              nickname,
              yearOfBirth,
              picture, //linkki googlen kuvahakuun
              weight,
              sport,
              record
            ) {
              super(firstnames, lastname, nickname, yearOfBirth);
              this.picture = picture;
              this._weight = weight;
              this.sport = sport;
              this.record = record;
            }
